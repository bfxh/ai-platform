258    def register(self, skill: Skill) -> bool:
259        """注册技能"""
260        if not skill.name:
261            self.logger.error("技能名称不能为空")
262            return False
263
264        if skill.name in self._skills:
265            self.logger.warning(f"技能 '{skill.name}' 已存在，将被覆盖")
266
267        # 初始化技能
268        if skill.initialize():
269            self._skills[skill.name] = skill
270            self._skill_classes[skill.name] = skill.__class__
271            self.logger.info(f"技能 '{skill.name}' 注册成功")
272            return True
273        else:
274            self.logger.error(f"技能 '{skill.name}' 初始化失败")
275            return False

277    def unregister(self, skill_name: str) -> None:
278        """注销技能"""

280            self._skills[skill_name].shutdown()
281            del self._skills[skill_name]
282            del self._skill_classes[skill_name]
283            self.logger.info(f"技能 '{skill_name}' 已注销")

293    def execute(self, skill_name: str, params: Dict[str, Any]) -> Dict[str, Any]:
294        """执行技能"""
295        skill = self._skills.get(skill_name)
296        if not skill:
297            return {"success": False, "error": f"技能 '{skill_name}' 不存在"}
298
299        # 执行前钩子
300        for hook in self._hooks["before_execute"]:
301            try:
302                hook(skill_name, params)
303            except Exception as e:
304                self.logger.error(f"执行前钩子错误: {e}")
305
306        # 验证参数
307        valid, error = skill.validate_params(params)
308        if not valid:
309            return {"success": False, "error": error}
310
311        # 执行技能
312        try:
313            skill.status = SkillStatus.RUNNING
314            self.logger.info(f"执行技能: {skill_name}")
315            result = skill.execute(params)
316            skill.status = SkillStatus.READY
317
318            # 执行后钩子
319            for hook in self._hooks["after_execute"]:
320                try:
321                    hook(skill_name, params, result)

323                    self.logger.error(f"执行后钩子错误: {e}")

325            self.logger.info(f"技能执行成功: {skill_name}")
326            return result

328        except Exception as e:
329            error_message = str(e)
330            skill._last_error = error_message
331            skill.status = SkillStatus.ERROR

334            for hook in self._hooks["on_error"]:

339

363    def _load_skill_from_dir(self, skill_dir: Path):
364        """从目录加载技能"""
365        skill_file = skill_dir / "skill.py"
366        if not skill_file.exists():
367            return
368
369        try:
370            # 动态导入
371            module_name = f"skills.{skill_dir.name}.skill"
372            spec = importlib.util.spec_from_file_location(module_name, skill_file)
373            module = importlib.util.module_from_spec(spec)
374            sys.modules[module_name] = module
375            spec.loader.exec_module(module)
376
377            # 查找技能类
378            for name, obj in inspect.getmembers(module):
379                if (
380                    inspect.isclass(obj)
381                    and issubclass(obj, Skill)
382                    and obj != Skill
383                    and hasattr(obj, "name")
384                    and obj.name
385                ):
386
387                    # 创建实例并注册
388                    skill_instance = obj()
389                    self.register(skill_instance)
390                    break
391
392        except Exception as e:
393            self.logger.error(f"加载技能失败 ({skill_dir.name}): {e}")
