 26class PerformanceMetric:
 27    """性能指标数据类"""
 28

 30    skill_name: str

 32    timestamp: datetime = field(default_factory=datetime.now)
 33    success: bool = True
 34    error_message: Optional[str] = None
 35
 36    def to_dict(self) -> Dict[str, Any]:
 37        """转换为字典"""

 48class PerformanceMonitor:
 49    """性能监控器"""
 50
 51    def __init__(self):
 52        self.metrics: List[PerformanceMetric] = []
 53        self.enabled = True
 54        self.max_metrics = 10000  # 最大保存的指标数量
 55
 56    def record_metric(self, metric: PerformanceMetric):
 57        """记录性能指标"""
 58        if not self.enabled:
 59            return
 60
 61        self.metrics.append(metric)
 62
 63        # 限制指标数量
 64        if len(self.metrics) > self.max_metrics:
 65            self.metrics = self.metrics[-self.max_metrics :]
 66
 67        # 记录慢操作
 68        if metric.execution_time > 1.0:  # 超过1秒认为是慢操作
 69            logger.warning(
 70                f"慢操作检测: {metric.skill_name}.{metric.function_name} "
 71                f"耗时 {metric.execution_time:.3f}s"
 72            )
 73
 74    def get_metrics(self, skill_name: Optional[str] = None) -> List[PerformanceMetric]:
 75        """获取性能指标"""
 76        if skill_name:
 77            return [m for m in self.metrics if m.skill_name == skill_name]
 78        return self.metrics.copy()
 79
 80    def get_average_time(self, skill_name: Optional[str] = None) -> float:
 81        """获取平均执行时间"""
 82        metrics = self.get_metrics(skill_name)
 83        if not metrics:
 84            return 0.0
 85        return sum(m.execution_time for m in metrics) / len(metrics)
 86
 87    def get_slowest_operations(self, limit: int = 10) -> List[PerformanceMetric]:
 88        """获取最慢的操作"""
 89        sorted_metrics = sorted(self.metrics, key=lambda x: x.execution_time, reverse=True)
 90        return sorted_metrics[:limit]
 91
 92    def get_skill_statistics(self) -> Dict[str, Dict[str, Any]]:
 93        """获取技能统计信息"""
 94        stats = defaultdict(
 95            lambda: {
 96                "count": 0,
 97                "total_time": 0.0,
 98                "avg_time": 0.0,
 99                "success_count": 0,
100                "error_count": 0,
101            }
102        )
103
104        for metric in self.metrics:
105            skill_stats = stats[metric.skill_name]
106            skill_stats["count"] += 1
107            skill_stats["total_time"] += metric.execution_time
108            if metric.success:
109                skill_stats["success_count"] += 1
110            else:
111                skill_stats["error_count"] += 1
112
113        # 计算平均值
114        for skill_name, skill_stats in stats.items():
115            if skill_stats["count"] > 0:
116                skill_stats["avg_time"] = skill_stats["total_time"] / skill_stats["count"]
117
118        return dict(stats)
119
120    def export_metrics(self, file_path: str):
121        """导出指标到文件"""
122        data = {
123            "export_time": datetime.now().isoformat(),
124            "total_metrics": len(self.metrics),
125            "metrics": [m.to_dict() for m in self.metrics],
126        }
127
128        with open(file_path, "w", encoding="utf-8") as f:
129            json.dump(data, f, ensure_ascii=False, indent=2)
130
131        logger.info(f"性能指标已导出到: {file_path}")
132
133    def clear_metrics(self):
134        """清空指标"""
135        self.metrics.clear()
136        logger.info("性能指标已清空")

148def monitor_performance(skill_name: Optional[str] = None):
149    """性能监控装饰器"""
150
151    def decorator(func: Callable) -> Callable:
152        @functools.wraps(func)
153        def wrapper(*args, **kwargs):
154            start_time = time.time()
155            error_message = None
156            success = True
157
158            try:
159                result = func(*args, **kwargs)
160                return result
161            except Exception as e:
162                success = False
163                error_message = str(e)
164                raise
165            finally:
166                execution_time = time.time() - start_time
167
168                # 获取技能名称
169                actual_skill_name = skill_name
170                if actual_skill_name is None and args:
171                    # 尝试从 self 获取技能名称
172                    self_obj = args[0]
173                    if hasattr(self_obj, "name"):
174                        actual_skill_name = self_obj.name
175
176                if actual_skill_name is None:
177                    actual_skill_name = "unknown"
178
179                # 记录性能指标
180                metric = PerformanceMetric(
181                    function_name=func.__name__,
182                    skill_name=actual_skill_name,
183                    execution_time=execution_time,
184                    success=success,
185                    error_message=error_message,
186                )
187
188                _performance_monitor.record_metric(metric)
189
190        return wrapper
191
192    return decorator

195class PerformanceProfiler:
196    """性能分析器"""
197
198    def __init__(self):
199        self.monitor = get_performance_monitor()
200
201    def analyze_bottlenecks(self) -> List[Dict[str, Any]]:
202        """分析性能瓶颈"""
203        bottlenecks = []
204
205        # 获取统计信息
206        stats = self.monitor.get_skill_statistics()
207
208        for skill_name, skill_stats in stats.items():
209            # 识别慢技能（平均执行时间超过1秒）
210            if skill_stats["avg_time"] > 1.0:
211                bottlenecks.append(
212                    {
213                        "type": "slow_skill",
214                        "skill_name": skill_name,
215                        "avg_time": skill_stats["avg_time"],
216                        "recommendation": "考虑优化技能执行逻辑或添加缓存",
217                    }
218                )
219
220            # 识别高错误率技能（错误率超过10%）
221            if skill_stats["count"] > 0:
222                error_rate = skill_stats["error_count"] / skill_stats["count"]
223                if error_rate > 0.1:
224                    bottlenecks.append(
225                        {
226                            "type": "high_error_rate",
227                            "skill_name": skill_name,
228                            "error_rate": error_rate,
229                            "recommendation": "检查错误处理逻辑和输入验证",
230                        }
231                    )
232
233        # 获取最慢的操作
234        slowest_ops = self.monitor.get_slowest_operations(5)
235        for op in slowest_ops:
236            if op.execution_time > 0.5:  # 超过0.5秒
237                bottlenecks.append(
238                    {
239                        "type": "slow_operation",
240                        "skill_name": op.skill_name,
241                        "function": op.function_name,
242                        "execution_time": op.execution_time,
243                        "recommendation": "优化此特定操作的性能",
244                    }
245                )
246
247        return bottlenecks
248
249    def generate_report(self) -> str:
250        """生成性能报告"""
251        lines = []
252        lines.append("=" * 60)
253        lines.append("性能监控报告")
254        lines.append("=" * 60)
255        lines.append("")
256
257        # 总体统计
258        lines.append("总体统计:")
259        lines.append(f"  总指标数: {len(self.monitor.metrics)}")
260        lines.append(f"  平均执行时间: {self.monitor.get_average_time():.3f}s")
261        lines.append("")
262
263        # 技能统计
264        lines.append("技能统计:")
265        stats = self.monitor.get_skill_statistics()
266        for skill_name, skill_stats in sorted(
267            stats.items(), key=lambda x: x[1]["avg_time"], reverse=True
268        ):
269            lines.append(f"  {skill_name}:")
270            lines.append(f"    调用次数: {skill_stats['count']}")
271            lines.append(f"    平均时间: {skill_stats['avg_time']:.3f}s")
272            lines.append(
273                f"    成功率: {(skill_stats['success_count'] / skill_stats['count'] * 100):.1f}%"
274                if skill_stats["count"] > 0
275                else "    成功率: N/A"
276            )
277        lines.append("")
278
279        # 瓶颈分析
280        lines.append("性能瓶颈:")
281        bottlenecks = self.analyze_bottlenecks()
282        if bottlenecks:
283            for bottleneck in bottlenecks:
284                lines.append(f"  [{bottleneck['type']}] {bottleneck.get('skill_name', 'N/A')}")
285                if "function" in bottleneck:
286                    lines.append(f"    函数: {bottleneck['function']}")
287                if "avg_time" in bottleneck:
288                    lines.append(f"    平均时间: {bottleneck['avg_time']:.3f}s")
289                if "error_rate" in bottleneck:
290                    lines.append(f"    错误率: {bottleneck['error_rate']:.1%}")
291                if "execution_time" in bottleneck:
292                    lines.append(f"    执行时间: {bottleneck['execution_time']:.3f}s")
293                lines.append(f"    建议: {bottleneck['recommendation']}")
294                lines.append("")
295        else:
296            lines.append("  未发现明显性能瓶颈")
297
298        lines.append("=" * 60)
299
300        return "\n".join(lines)
